void RCC_Config()
{
RCC-> |=0X00010000;
while(!(RCC->CR&0X00020000));
RCC->CR|=0X00080000;
RCC->PLLCFGR|=0X00040000;
RCC->PLLCFGR|=0X00000004;
RCC->PLLCFGR|=0X00015A00;
RCC->PLLCFGR|=0X00010000;
RCC->CFGR|=0X00010000;
RCC->CFGR|=0X00080000;
RCC->CFGR|=0X00001400;
RCC->CIR|=0X00080000;
CC->CIR|=0X00008000;
}
